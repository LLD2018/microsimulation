497  copy(v.begin(), v.end(), inserter(_partition, _partition.begin()));
498  }
499  void clear() {
500  _ut.clear();
501  _pt.clear();
502  _events.clear();
503  _prev.clear();
504  _partition.clear();
505  _vector.clear();

507  Utility discountedUtilities(Time a, Time b, Utility utility = 1.0) {
508  if (discountRate == 0.0) return utility * (b-a);
509  else if (a==b) return 0.0;
510  else if (discountRate>0.0) {
511  double alpha = log(1.0+discountRate);
512  return utility/alpha*(exp(-a*alpha) - exp(-b*alpha));
513  }
514  else {
515  REprintf("discountRate less than zero.");
516  return 0.0;
517  }
518  }
519  void add(const State state, const Event event, const Time lhs, const Time rhs, const Utility utility = 1, int index = 0) {
520  Iterator lo, hi, it, last;
521  lo = _partition.lower_bound(lhs);
522  hi = _partition.lower_bound(rhs);
523  last = _partition.begin(); // because it is ordered by greater<Time>
524  ++_events[boost::make_tuple(state,event,*hi)];
525  bool iterating;
526  for(it = lo, iterating = true; iterating; iterating = (it != hi), --it) { // decrement for greater<Time>
527  if (lhs<=(*it) && (*it)<rhs) // cadlag
528  ++_prev[Pair(state,*it)];
529  if (it == last) {
530  if (outputUtilities) {
531  Utility u = discountedUtilities(std::max<Time>(lhs,*it), rhs, utility);
532  _ut[Pair(state,*it)] += u;
533  _vector[index] += u;
534  }
535  _pt[Pair(state,*it)] += rhs - std::max<Time>(lhs,*it);
536  }
537  else {
538  Time next_value = *(--it); it++; // decrement/increment for greater<Time>
539  if (outputUtilities) {
540  Utility u = discountedUtilities(std::max<Time>(lhs,*it), std::min<Time>(rhs,next_value), utility);
541  _ut[Pair(state,*it)] += u;
542  _vector[index] += u;
543  }
544  _pt[Pair(state,*it)] += std::min<Time>(rhs,next_value) - std::max<Time>(lhs,*it);
545  }
546  }
